# To run LAG unit-test do net steps:
## 1. Build SAI
cd stub
./autogen.sh
./configure --prefix=/usr
make && sudo make install
## 2. Compile lag_test.c
cd SAI/test/sai_ut
gcc -o lag_unit_test -I /usr/include/sai -L /usr/lib lag_test.c -lsai 
## 3. Run test
./lag_unit_test
